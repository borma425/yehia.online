from html.entities import name2codepoint

class Yehia(HTMLParser):
    def handle_starttag(self, tag, attrs):
        print("Start of tag :", tag)

    def handle_endtag(self, tag):
        print("End of tag :", tag)

    def handle_data(self, data):
        print("Data     :", data)

    def handle_comment(self, data):
        print("Comment  :", data)

Online = Yehia()
Online.feed(
# First Line
    '<script type="text/javascript">'
#Line2 Run Def
             'alert("Hello Yehia");</script>'
# Line 3 Run Def
             '<!--Write a comment to me-->')
# Single line Run Def
Online.feed('<style>#Yehia{font-size:14px;}</style>')
